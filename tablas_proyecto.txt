create table patrocinador(
	ID int primary key,
	nombre varchar(60),
	producto varchar(60)
);

create table ciudad(
	codigo int primary key, 
	nombre varchar(60)
);

create table artista(
	ID int primary key,
	nombre varchar (60),
	integrantes int,
	genero varchar (60),
	manager varchar (60)
);

create table empresa_promotora(
	ID int primary key,
	nombre varchar (60)
);

create table centro_de_eventos(
	codigo int primary key,
	nombre varchar(60),
	direccion varchar(60),
	capacidad int,
	codigo_ciudad int,
	foreign key (codigo_ciudad) references ciudad
);

create table evento (
	ID int primary key,
	nombre varchar(60) not null,
	fecha_inicio date,
	fecha_fin date,
	codigo_Centro_de_Eventos int,
	foreign key (codigo_Centro_de_Eventos) references centro_de_eventos
);

create table boleteria(
	ID int primary key,
	codigo_boleteria int,
	ID_evento int,
	cantidad int,
	precio_max int,
	precio_min int,
	foreign key (codigo_boleteria) references empresa_promotora,
	foreign key (ID_evento) references evento
);

create table asiste(
	ID_artista int,
	ID_evento int,
	foreign key (ID_artista) references artista,
	foreign key (ID_evento) references evento
);

create table patrocina(
	ID_patrocinador int,
	ID_evento int,
	foreign key (ID_patrocinador) references patrocinador,
	foreign key (ID_evento) references evento
);
